#version 330 core

layout (location = 0) in vec2 vertexPos;
layout (location = 1) in vec2 vertexUV;

uniform vec2 uPos;
uniform vec2 uSize;
uniform mat4 uProjection;

out vec2 vUV;

void main() {
    vec2 worldPos = uPos + vertexPos * uSize;
    gl_Position = uProjection * vec4(worldPos, 0.0, 1.0);
    vUV = vertexUV;
}
